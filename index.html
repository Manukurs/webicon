<i8-icon icon="symbol:about"></i8-icon>
<span i8-icon="g:home"></span>
<div data-i8-icon="icons8_cup" alt="Icons8 Cup"></div>
$(function() {
  $(document).i8icons({
    iconSets: {
      symbol: 'assets/icon-sets/symbol/icon-set.svg',
      g: 'assets/icon-sets/group/icon-set.svg'
    }
  });
});
<i8-icon icon="symbol:about"></i8-icon>
<span i8-icon="g:home"></span>
<div data-i8-icon="icons8_cup" alt="Icons8 Cup"></div>
angular.module('app')
  .config(function($i8IconProvider) {
    $i8IconProvider
      .iconSet('symbol', 'assets/icon-sets/symbol/icon-set.svg')
      .iconSet('g', 'assets/icon-sets/group/icon-set.svg')
    ;
  });

<div data-i8-icon="f:home-ios" data-alt="home" style="color: #2980b9;"></div>
$(function () {
  $(document).i8icons({
    fonts: {
      f: 'custom-iconic-font custom-iconic-font-?'
    }
  });
});
