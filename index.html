Load Svg Icons From Url

  • AngularJS
  • jQuery
<i8-icon icon="like"></i8-icon>
<span i8-icon="department"></span>
<div data-i8-icon="assets/icons/svg/doughnut_chart.svg" data-alt="doughnut_chart"></div>
$(function() {
  $(document).i8icons({
    icons: {
      like: 'assets/icons/svg/like.svg',
      department: 'assets/icons/svg/department.svg'
    }
  });
});
<i8-icon icon="like"></i8-icon>
<span i8-icon="department"></span>
<div data-i8-icon="assets/icons/svg/doughnut_chart.svg" data-alt="doughnut_chart"></div>
angular.module('app')
  .config(function($i8IconProvider) {
    $i8IconProvider
      .icon('like', 'assets/icons/svg/like.svg')
      .icon('department', 'assets/icons/svg/department.svg')
    ;
  });

Font Icons

  • AngularJS
  • jQuery
Display 3 font-icons, each with different colors.
<i8-icon icon="fa:home" style="color: #d35400;"></i8-icon>
<span i8-icon="gi:home" style="color: #2c3e50;"></span>
<div data-i8-icon="f:home-ios" data-alt="home" style="color: #2980b9;"></div>
$(function () {
  $(document).i8icons({
    fonts: {
      f: 'custom-iconic-font custom-iconic-font-?'
    }
  });
});
