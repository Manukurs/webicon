Font Awesome icons



<i8-icon icon="fa:heart 3x" class="heart"></i8-icon>
<span class="i8-icon i8-icon-fa:home|3x"></span>
<i8-icon icon="fa:circle-o-notch spin 2x" id="spin"></i8-icon>

<script>
  $(function() {
    var
      spinners = ['circle-o-notch', 'cog', 'refresh', 'spinner'],
      index = 0;
    setInterval(function() {
      $('#spin')
        .attr('icon', 'fa:' + spinners[++index % spinners.length] + ' spin 2x')
        .i8icon();
    }, 1500)
  });
</script>

<style>
  .heart {
    color: #F44336;
  }
</style>




<i8-icon icon="fa:heart 3x" class="heart"></i8-icon>
<span i8-icon="fa:home 3x"></span>
<i8-icon icon="{{spinner}} spin 2x"></i8-icon>

<script>
  angular
    .module('app', ['i8.icon'])
    .run(function($rootScope, $interval) {
      var
        spinners = ['fa:circle-o-notch', 'fa:cog', 'fa:refresh', 'fa:spinner'],
        index = 0;
      $rootScope.spinner = spinners[0];
      $interval(function() {
        $rootScope.spinner = spinners[++index % spinners.length];
      }, 1500);
    });
</script>

<style>
  .heart {
    color: #F44336;
  }
</style>
